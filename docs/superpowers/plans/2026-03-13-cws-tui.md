# cws-tui Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `cws-tui`, a Go/bubbletea terminal UI that replaces the fzf navigator in `clode-ws.sh` with a persistent split-view workspace manager.

**Architecture:** A single Go binary renders both a left tree pane (projects → worktrees → terminals) and a right live-preview pane using bubbletea + lipgloss. It reads state from tmux (`list-sessions`, `list-windows`) and Docker (`docker ps`) every 2 seconds and writes state transitions back via tmux commands and shell invocations to clode-ws.sh helpers.

**Tech Stack:** Go 1.22+, [bubbletea v0](https://github.com/charmbracelet/bubbletea), [lipgloss v0](https://github.com/charmbracelet/lipgloss), tmux, Docker, zsh

**Spec:** `docs/superpowers/specs/2026-03-13-cws-tui-design.md`

package main

import (
    "fmt"
    "os"

    tea "github.com/charmbracelet/bubbletea"
)

type model struct{ quitting bool }

func (m model) Init() tea.Cmd { return nil }

func (m model) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
    if _, ok := msg.(tea.KeyMsg); ok {
        m.quitting = true
        return m, tea.Quit
    }
    return m, nil
}

func (m model) View() string {
    if m.quitting {
        return ""
    }
    return "cws-tui loading... (press any key)\n"
}

func main() {
    p := tea.NewProgram(model{}, tea.WithAltScreen())
    if _, err := p.Run(); err != nil {
        fmt.Fprintf(os.Stderr, "error: %v\n", err)
        os.Exit(1)
    }
}
```

// Inject a fake in tests; use RealRunner in production.
type Runner func(args ...string) (string, error)

// RealRunner calls the real tmux binary using os/exec.Command with separate args
// (no shell string construction — avoids shell injection).
func RealRunner(args ...string) (string, error) { ... }

// Window represents one tmux window.
type Window struct {
    Index int
    Name  string  // e.g. "main:host-1"
}

// Client wraps tmux subcommands.
type Client struct { run Runner }

func NewClient(run Runner) *Client { ... }

// ListSessions returns all tmux session names (empty slice if no sessions).
func (c *Client) ListSessions() ([]string, error) { ... }

// ListWindows returns windows in a session, parsed from "index name" format.
func (c *Client) ListWindows(session string) ([]Window, error) { ... }

// CapturePane returns the last 50 lines of the window at the given index.
// Resolves by index (not name) to avoid ":" ambiguity in window names.
func (c *Client) CapturePane(session string, windowIndex int) (string, error) { ... }

// SwitchClient switches the tmux client to "session:index".
func (c *Client) SwitchClient(target string) error { ... }

// DetachClient detaches the current tmux client (keeps session alive).
func (c *Client) DetachClient() error { ... }

// NewSession creates a new detached tmux session.
func (c *Client) NewSession(name, startDir, windowName string) error { ... }

// HasSession reports whether a session exists.
func (c *Client) HasSession(name string) bool { ... }
```

Implementation notes:
- `RealRunner`: use `os/exec.Command("tmux", args...).Output()` — args are separate, no shell involved
- `ListSessions`: format flag `-F #{session_name}`; return nil (not error) when tmux has no sessions (exit code 1)
- `ListWindows`: format flag `-F "#{window_index} #{window_name}"`; parse with `strings.SplitN(line, " ", 2)`
- `CapturePane`: target is `fmt.Sprintf("%s:%d", session, windowIndex)`; flags `-p -e -S -50`

type Status int

    Name       string
    Dir        string // absolute disk path; empty if disk-only and not found

    Worktrees  []Worktree
}

type State struct {
    Projects []Project

// fakeTmuxRunner dispatches based on the first argument (the tmux subcommand).
// windowsBySession maps session name → "index name\nindex name\n..." output.
func fakeTmuxRunner(sessions string, windowsBySession map[string]string) tmux.Runner {
    return func(args ...string) (string, error) {
        switch args[0] {
        case "list-sessions":
            return sessions, nil
        case "list-windows":
            // args pattern: list-windows -t <session> -F ...
            // find session name (the arg after "-t")
            for i, a := range args {
                if a == "-t" && i+1 < len(args) {
                    return windowsBySession[args[i+1]], nil
                }
            }
            return "", nil
        case "has-session":
            // args pattern: has-session -t <session>
            for i, a := range args {
                if a == "-t" && i+1 < len(args) {
                    name := args[i+1]
                    for _, s := range strings.Split(sessions, "\n") {
                        if strings.TrimSpace(s) == name {
                            return "", nil // exists → nil error
                        }
                    }
                    return "", os.ErrProcessDone // not found → error
                }
            }
            return "", os.ErrProcessDone
        }
        return "", nil
    }
}

func fakeDockerRunner(runningContainerNames string) tmux.Runner {
    return func(args ...string) (string, error) {
        return runningContainerNames, nil
    }
}

// Arguments are shell-quoted with fmt.Sprintf("%q", arg) before interpolation.
func shellHelperCmd(fn string, args ...string) tea.Cmd {
    return func() tea.Msg {
        cwsScript := filepath.Join(os.Getenv("HOME"), "Projects/clode/clode-ws.sh")
        // Build: . "/path/to/clode-ws.sh" && fn "arg1" "arg2"
        parts := fmt.Sprintf(". %s && %s", shellQuote(cwsScript), fn)
        for _, a := range args {
            parts += " " + shellQuote(a)
        }
        // os/exec with zsh -c <quoted-string>: zsh interprets the string, but
        // all user-controlled values are shell-quoted so no injection is possible.
        cmd := exec.Command("zsh", "-c", parts)
        cmd.Run()
        return refreshMsg{}
    }
}

func shellQuote(s string) string {
    return "'" + strings.ReplaceAll(s, "'", "'\\''") + "'"
}

# cws — launch the cws-tui workspace manager
cws() {
  if ! command -v cws-tui &>/dev/null; then
    echo "cws-tui not found — run: make -C ~/Projects/clode install" >&2
    return 1
  fi
  local prefix
  prefix=$(tmux show-option -gv prefix 2>/dev/null)
  if [[ "$prefix" == "C-a" ]]; then
    echo "warning: tmux prefix is Ctrl+A — action mode (Ctrl+A) will not work." >&2
    echo "Change tmux prefix to Ctrl+B or set _CLODE_WS_ACTION_KEY. See clode-ws.sh." >&2
  fi
  if tmux has-session -t cws-ui 2>/dev/null; then
    tmux attach-session -t cws-ui
  else
    tmux new-session -d -s cws-ui -n cws-panel
    tmux send-keys -t cws-ui:cws-panel "cws-tui" Enter
    tmux attach-session -t cws-ui
  fi
